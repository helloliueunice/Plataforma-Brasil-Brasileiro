        function mostrarEditais(editais) {

            var lista = document.getElementById('listaEditais');

            var html = '';

            if (editais.length === 0) {

                html = '<div class="edital-card"><p>Nenhum edital</p></div>';

            } else {

                for (var i = 0; i < editais.length; i++) {

                    var e = editais[i];

                    html += '<div class="edital-card">';

                    html += '<h3>' + e.titulo + '</h3>';

                    html += '<p><strong>' + e.orgao + '</strong></p>';

                    html += '<p>' + e.descricao + '</p>';

                    html += '<div><span class="tag">' + e.tag + '</span>';

                    html += '<span class="tag privado">' + e.tipo + '</span></div>';

                    html += '<div class="edital-info">';

                    html += '<div><small>Min</small><br><strong>' + e.min + '</strong></div>';

                    html += '<div><small>Max</small><br><strong>' + e.max + '</strong></div>';

                    html += '<div><small>Prazo</small><br><strong>' + e.prazo + '</strong></div>';

                    html += '<div><a href="' + e.link + '" target="_blank" class="link-btn">Acessar</a></div>';

                    html += '</div></div>';

        function filtrarEditais() {

            var busca = document.getElementById('inputBusca').value.toLowerCase();

            var filtrados = [];

            for (var i = 0; i < todosEditais.length; i++) {

                var e = todosEditais[i];

                var passaTag = tagSelecionada === 'todos' || e.tag === tagSelecionada;

                var passaBusca = busca === '' || e.titulo.toLowerCase().indexOf(busca) >= 0 || e.orgao.toLowerCase().indexOf(busca) >= 0;

                if (passaTag && passaBusca) {

                    filtrados.push(e);

                }

            }

            mostrarEditais(filtrados);

        }


        function filtrarPorTag(tag) {

            tagSelecionada = tag;

            var botoes = document.querySelectorAll('.filter-btn');

            for (var i = 0; i < botoes.length; i++) {

                if (botoes[i].getAttribute('data-tag') === tag) {

                    botoes[i].className = 'filter-btn active';

                } else {

                    botoes[i].className = 'filter-btn';

                }

            }

            filtrarEditais();

        }


        function mudarPagina(num) {

            for (var i = 1; i <= 5; i++) {

                document.getElementById('pagina' + i).className = 'page';

                document.getElementById('btn' + i).className = '';

            }

            document.getElementById('pagina' + num).className = 'page active';

            document.getElementById('btn' + num).className = 'active';

        }
