        // Funções de navegação

        function mostrarPagina(pagina) {

            document.getElementById('pagina-inicio').style.display = 'none';

            document.getElementById('pagina-editais').style.display = 'none';

            document.getElementById('pagina-dashboard').style.display = 'none';

            document.getElementById('pagina-' + pagina).style.display = 'block';

            

            if (pagina === 'dashboard') {

                criarGrafico();

            }

        }


        function filtrarEditais() {

            const busca = document.getElementById('busca').value.toLowerCase();

            const lista = document.getElementById('lista-editais');

            const itens = lista.querySelectorAll('div[class*="border-gray-200"]');

            

            itens.forEach(item => {

                const texto = item.innerText.toLowerCase();

                item.style.display = texto.includes(busca) ? 'block' : 'none';

            });

        }


        function criarGrafico() {

            const ctx = document.getElementById('chartValores');

            if (ctx && !window.chartJaFoiCriado) {

                new Chart(ctx, {

                    type: 'bar',

                    data: {

                        labels: ['Saúde', 'Educação', 'Infraestrutura', 'Meio Ambiente'],

                        datasets: [{

                            label: 'Valores (R$ Mil)',

                            data: [2500, 1800, 4200, 3100],

                            backgroundColor: '#006B9F'

                        }]

                    },

                    options: {

                        responsive: true,

                        plugins: {

                            legend: {

                                display: false

                            }

                        }

                    }

                });

                window.chartJaFoiCriado = true;

            }

        }
